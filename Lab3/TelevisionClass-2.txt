//Declare class
Class Televison
      
	//Declare class variables
	String manufacturer
	Numberic screenSize
	Numeric price
      
	//Constructor
	Constructor Method Television(String mName, Numeric screenS, Numeric cost)
        	manufacturer = mName
        	screenSize = screenS
        	price = cost
	End Method
      
	//Set Price
	Method setPrice(Numeric newPrice)
        	price = newPrice
	End Method
      
	//Display Info
	Method displayInfo()
        	Display "Manufacturer is: " + manufacturer + ". Screen Size is: " + screenSize + " inches. Price is: $" + price + "."
      End Method
End Class



Start
	Television testTV = new Television("Sony", 52, 1299.00)
	testTV.displayInfo()
	testTV.setPrice(999.00)
	testTV.displayInfo()
Stop
